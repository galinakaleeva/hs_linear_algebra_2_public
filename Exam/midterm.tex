Linear Algebra 2. Harbour.Space University

Midterm exam
\end{center}

\begin{enumerate}
\item \begin{enumerate}[(a)]
    \item Prove that the polynomials $1,\, x-1,\, (x-1)^2,\, (x-1)^3$ form a basis of $\R_3[x]$.
    \item Consider the map $\varphi\colon \R_3[x] \to \R_3[x]$ defined by $\varphi(p(x)) = (x-1)^2\, p''(x)$. Prove that it is a linear operator on $\R_3[x]$.
    \item Find the matrix of $\varphi$ in the basis $1,\, x-1,\, (x-1)^2,\, (x-1)^3$.
    \item Compute $\varphi\!\left(1 + (x-1) + \frac{(x-1)^2}{2} + \frac{(x-1)^3}{6}\right)$.
    \item Find $\tr(\varphi)$ and $\det(\varphi)$.
\end{enumerate}

    \item In some basis $e_1, e_2, e_3$ a linear operator has matrix
    $$
    A=\begin{pmatrix}
    0 & 1 & 1 \\
    -2 & 3 & 1 \\
    -3 & 1 & 4
    \end{pmatrix}.
    $$
    Find its matrix in the basis
    $$
    \tilde{e}_1=e_1+e_2+e_3,\quad \tilde{e}_2=e_1+2 e_2+ e_3,\quad \tilde{e}_3=e_1+e_2+2e_3.
    $$

\item \begin{enumerate}[(a)]
    \item Find the eigenvalues of the operator given in some basis by the matrix
    $$
    A = \begin{pmatrix}
    3 & 2 & 1 \\
    0 & 2 & 1 \\
    0 & 0 & 1
    \end{pmatrix}.
    $$
    \item Does there exist a basis in which the matrix of the operator is diagonal? Justify your answer.
\end{enumerate}

\item Compute $A^{2025}$, where 
$$
A = \begin{pmatrix}
    -5 & 13 \\
    -2 & 5\\
\end{pmatrix}.
$$

\item Find a closed formula for the solution of the following recurrence relation:
$$
u(n)=4 u(n-1)-3 u(n-2), \quad u(0)=1, \quad u(1)=3.
$$

\end{enumerate}
