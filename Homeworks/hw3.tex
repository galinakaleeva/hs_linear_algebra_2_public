\begin{enumerate}[1. ]

\item Let $A$ be an $m \times n$ matrix of rank $n$. Prove that $A^{+}=\left(A^T A\right)^{-1} A^T$. (Beware non-invertible matrices! Don't try to invert them!)

\item Find SVD of the matrix $A$. What is the rank of $A$? Find the best low-rank approximation of $A$ of rank $r-1$, where $r = \rk A$:
$$
A = \begin{pmatrix}
    2 & 1 & 0 & -1 \\
    0 & -1 & 1 & 1
    \end{pmatrix}.
$$

\item Consider two polinomials $p(x) = x^2 + 1$ and $q(x) = x^3 + x + 1$. Let us mupliply them using the \st{fast} Fourier transform. 

\begin{enumerate}[1) ]
\item Since $\deg (pq) = 5$, we need 6 points to recover the polynomial $pq$ by the FFT. Compute the Fourier transform of $p(x)$ and $q(x)$:
$$
y_p = F_6 c_p, y_q = F_6 c_q,
$$
where $c_p = (1, 0, 1, 0, 0, 0)^T$ and $c_q = (1, 1, 0, 1, 0, 0)^T$ are the coefficients of $p(x)$ and $q(x)$ respectively (we need length 6 because of the degree of the resulting polynomial).

\item Find $y$ as a pointwise product of $y_p$ and $y_q$.

\item Knowing $y$, find the coefficients of $pq$ from the equation:
$$
y = F_6 c,
$$
where $c = (c_0, c_1, c_2, c_3, c_4, c_5)^T$ are the coefficients of $pq(x)$.
\end{enumerate}

Explain why this algorithm gives the correct result.

\end{enumerate}
